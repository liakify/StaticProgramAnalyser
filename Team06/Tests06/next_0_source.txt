    y = 6;
    print x;
    if (x < y) then {
        if (y > 0) then {
            read y;
            print x; } else {
            x = y; }
        call Next2; } else {
        while (x > 0) {
            x = x;  }
        print y;  }  
    x = x;  }

        while (x > y) {
            print x; }
        if (x == y) then {
            read x; } else {
            call Next3;  }
