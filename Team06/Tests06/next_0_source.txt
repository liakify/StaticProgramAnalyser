procedure SelectNext {
    x = 5;
    y = 6;
    print x;
    if (x < y) then {
        read x;
        if (y > 0) then {
            read y;
            print x; } else {
            x = y; }
        call Next2; } else {
            while (x > 0) {
                x = x;    }  }  }
procedure Next2 {
    while (x < y) {
        x = x + y;
        while (x > y) {
            print x;
            read y;  }  }  }
